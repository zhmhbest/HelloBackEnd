# SpringMVCDemo

```bash
# 创建项目
ProjectName=SpringMVCDemo
mvn archetype:generate \
    "-DgroupId=org.example" \
    "-DartifactId=${ProjectName}" \
    "-DarchetypeArtifactId=maven-archetype-webapp" \
    "-DinteractiveMode=false"
cd ${ProjectName}
mkdir "src/main/java"
mkdir "src/main/webapp/WEB-INF/pages"
```
